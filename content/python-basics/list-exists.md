```python
def separate_even_odd(lst):
    even = []
    odd = []
    for num in lst:
        if num % 2 == 0:
            even.append(num)
        else:
            odd.append(num)
    print("Even Numbers:", even)
    print("Odd Numbers:", odd)

# Example usage
separate_even_odd([1, 2, 3, 4, 5, 6])
```

    Odd Numbers: [1, 3, 5]
