sum = 0
remainder = 0

    remainder = num % 10
    sum = sum + remainder
    num = int(num / 10)

# Program to find the Fibonacci series of numbers till 30.
num1 = 0
num2 = 1
print('Fibonacci series of numbers till 30 are: ')
print(num1, num2, end = ' ')
while num2 < 21:
    num1, num2 = num2, num1 + num2
    print(num2, end = ' ')
