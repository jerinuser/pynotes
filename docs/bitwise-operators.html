# Bitwise Operators in Python

# &     Bitwise AND                 Binary
# |     Bitwise OR                  Binary
# ^     Bitwise XOR (Exclusive OR)  Binary
# ~     Bitwise NOT                 Unary
# <<    Bitwise Left Shift          Binary
# >>    Bitwise Right Shift         Binary

B = 6
result = A & B

print("Result of (2 & 6): ", result)
