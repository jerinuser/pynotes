def second_largest(nums):
    nums.remove(max(nums))
    return max(nums)

# Test the function
nums = [10, 20, 30, 40, 50]
print("Second largest number:", second_largest(nums))

def separate_even_odd(lst):
    even = []
    odd = []
    for num in lst:
        if num % 2 == 0:
            even.append(num)
        else:
            odd.append(num)
    print("Even Numbers:", even)
    print("Odd Numbers:", odd)

# Example usage
separate_even_odd([1, 2, 3, 4, 5, 6])

Even Numbers: [2, 4, 6]
Odd Numbers: [1, 3, 5]

def is_list_palindrome(lst):
    start = 0
    end = len(lst) - 1
    while start < end:
        if lst[start] != lst[end]:
            print("The list is not a palindrome.")
            return
        start += 1
        end -= 1
    print("The list is a palindrome.")

# Example usage

def move_zeroes(lst):
    non_zero = [x for x in lst if x != 0]
    zeroes = [0] * lst.count(0)
    result = non_zero + zeroes
    print("List after moving zeroes:", result)

        self.length = length

    def __add__(self, other):
