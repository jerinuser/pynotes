def separate_even_odd(lst):
    even = []
    odd = []
    for num in lst:
        if num % 2 == 0:
            even.append(num)
        else:
            odd.append(num)
    print("Even Numbers:", even)
    print("Odd Numbers:", odd)

# Example usage
separate_even_odd([1, 2, 3, 4, 5, 6])

def is_list_palindrome(lst):
    start = 0
    end = len(lst) - 1
    while start < end:
        if lst[start] != lst[end]:
            print("The list is not a palindrome.")
            return
        start += 1
        end -= 1
    print("The list is a palindrome.")

# Example usage

    non_zero = [x for x in lst if x != 0]
    zeroes = [0] * lst.count(0)

def count_numbers(lst):
    positives = 0
    negatives = 0
    zeros = 0
    for num in lst:
        if num > 0:
            positives += 1
        elif num < 0:
            negatives += 1
        else:
            zeros += 1
    print(f"Positive: {positives}, Negative: {negatives}, Zero: {zeros}")

# Example usage
count_numbers([1, -2, 0, 4, -5, 0])

def is_prime(num):
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

# Example usage
print(is_prime(7))  # True
print(is_prime(10)) # False
