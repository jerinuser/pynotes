def check_palindrome(input_string):
    # Check if the string is a palindrome (ignoring case)
    is_palindrome = input_string.lower() == input_string[::-1].lower()

    if is_palindrome:
        return f"{input_string} is a palindrome"
    else:
        return f"{input_string} is not a palindrome"

# Test the function

def count_numbers(lst):
    positives = 0
    negatives = 0
    zeros = 0
    for num in lst:
        if num > 0:
            positives += 1
        elif num < 0:
            negatives += 1
        else:
            zeros += 1
    print(f"Positive: {positives}, Negative: {negatives}, Zero: {zeros}")

# Example usage

def is_prime(num):
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

# Example usage
print(is_prime(7))  # True
print(is_prime(10)) # False

def count_vowels(s):
    vowels = "aeiouAEIOU"
    count = 0
    i = 0
    while i < len(s):
        if s[i] in vowels:
            count += 1
        i += 1
    print("Number of vowels:", count)

# Example usage
count_vowels("hello world")
