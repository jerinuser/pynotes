# global Keyword

g_var = 50
def func_read():
    print(g_var)
def func_write():
    global g_var
    g_var = 100

func_read()
func_write()
func_read()

50
100

#  nonlocal Keyword

def outer_func():
    x = 50
    def inner_func():
        nonlocal x
        x = 100
        print("Inner function: ",x)
    inner_func()
    print("Outer function: ",x)

outer_func()

Inner function:  100
Outer function:  100
