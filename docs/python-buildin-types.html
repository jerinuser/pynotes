# Tuple Data type
#

t = (10, 20, "Python", 2 + 10j)
print(t)
print(type(t))

(10, 20, 'Python', (2+10j))
<class 'tuple'>

# Set Data type

s = {1, 2, 'Hello', 4 + 50j}
print(s)
print(type(s))

{1, 'Hello', 2, (4+50j)}
<class 'set'>

# Dictionary Data type

dict = {1 : 'Orange',
        2 : 'Apple',
        3 : 'Banana'}

print(dict)
print(type(dict))

{1: 'Orange', 2: 'Apple', 3: 'Banana'}
<class 'dict'>
