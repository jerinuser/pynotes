# Function with a Docstring.
def getSum(x, y):
    """This is a docstring

    This method takes two argument values and returns the summation of it."""
    print("Sum = ", (x + y))
getSum(20, 40) # Calling function.
print(getSum.__doc__) # Calling __doc__ method to print docstring.
